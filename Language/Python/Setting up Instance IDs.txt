
####An example of how to set makeshift IDs for each individual instance of the class. Inheritable.

class ExampleClass:    

	####Object ID Methods

    numObj = 0;

    @classmethod
    def getObjNum(cls):
        return cls.numObj;

    @classmethod
    def incrementObjNum(cls):
        cls.numObj += 1;

    @classmethod
    def getClassName(cls):
        return cls.__name__;
    
    def setObjID(self):
        self.ID = str(self.getClassName()) + str(self.getObjNum());
        self.incrementObjNum();

    def getObjID(self):
        return self.ID;
		
	def __init__(self):
		self.setObjID();
